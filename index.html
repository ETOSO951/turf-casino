// --- QRコード生成 --- function generateQRCode(roomName) { const canvas = document.getElementById('qr-code'); // canvasのサイズを明示的に設定（重要！） canvas.width = 280; canvas.height = 280; // スマホからアクセスするためのURL const baseUrl = window.location.origin + window.location.pathname; const token = 'tok_' + Math.random().toString(36).slice(2, 11) + Date.now().toString(36); const qrUrl = `${baseUrl}?room=${encodeURIComponent(roomName)}&token=${token}&type=mobile-guest`; console.log('Generating QR for:', qrUrl); console.log('Canvas size:', canvas.width, 'x', canvas.height); // QRコード生成 - Promise形式で正しく処理 QRCode.toCanvas(canvas, qrUrl, { errorCorrectionLevel: 'H', type: 'image/png', width: 250, margin: 1, color: { dark: '#000000', light: '#FFFFFF' } }).then(() => { console.log('QRコード生成成功'); }).catch(err => { console.error('QRコード生成エラー:', err); }); // トークンを保存（検証用） if(!window.validTokens) window.validTokens = {}; window.validTokens[token] = { room: roomName, type: 'mobile-guest', createdAt: Date.now() }; }
